ALTER PROCEDURE [dbo].[pr_GD_CHOT_LAI_DETAIL_SelectByIdChotLai]
@ID_CHOT_LAI NUMERIC(18,0)
AS
SELECT
	[ID],
	[ID_CHOT_LAI],
	[ID_TRAI_CHU],
	[SO_LUONG_TINH_LAI],
	[SO_TIEN_LAI],
	[DA_NHAN_TIEN_YN],
	[NGAY_NHAN_TIEN]
FROM [dbo].[GD_CHOT_LAI_DETAIL]
WHERE [ID_CHOT_LAI] = @ID_CHOT_LAI
OR @ID_CHOT_LAI = -1
ORDER BY 
	[ID] ASC
